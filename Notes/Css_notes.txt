Color :
  color : name , hex code , rgb(0,0,0)
  background-color : name , hex code , rgb(0,0,0)
  opacity : transparency , from 0.0 to 1 or 0%
  rgba : the normal rgb + opacity
  Hue : color by degree, from 0 to 360
  Saturation : by 0%
  Lightness : 0% white , 100% black
  hsl / hsla

Text :
  font-family : name
  font-size : 0px , 0% , 0em (width)
  @font-face : to use a font, even if it is not installed on the computer of the person. by font-family and src.
  font-weight : normal, bold
  font-style : normal , italic , oblique
  text-transform : uppercase , lowercase , capitalize.
  text-decoration : none , underline , overline , line-through , blink
  line-height : the space between each line , 0em
  letter-spacing : the space between each letter , 0em
  word-spacing : the space between each word , 0em
  text-align : left , right , center , justify
  vertical-align : baseline , sub , super , top , text-top , middle , bottom
  text-indent : 0px
  :first-letter : to apply special property to the first letter in the line
  :first-line : to apply special property to the first line
  :line : to set properties for links which haven't been visited yet
  :visited : to set properties for links which haven't been visited yet
  :hover : to set properties when cursor is brought on it
  :active : to set properties when an element is being active like a button
